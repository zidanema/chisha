```python
def meal_idx_to_slot(idx: int) -> tuple[Literal["lunch","dinner"], int]:
    """idx 0-based → (meal_type, day_index 1-based).
    
    idx=0 → ("lunch", 1)
    idx=1 → ("dinner", 1)
    idx=2 → ("lunch", 2)
    ...
    """
    if idx < 0:
        raise ValueError(f"meal_idx must be >= 0, got {idx}")
    day = idx // 2 + 1
    meal_type = "lunch" if idx % 2 == 0 else "dinner"
    return (meal_type, day)


def advance_meal(*, sid: str | None = None, root: Path | None = None) -> dict:
    """current_meal_idx++, dinner→next 时 current_date += 1 day + day_index++.
    
    state.history 由调用方写 (本函数只动 clock).
    """
    with _STATE_LOCK:
        s = state(sid=sid, root=root)
        if not s.get("enabled"):
            raise RuntimeError(...)
        cur_idx = int(s.get("current_meal_idx", 0))
        new_idx = cur_idx + 1
        if new_idx > int(s.get("total_meals", 14)):
            raise RuntimeError(f"meal_idx {new_idx} exceeds total {s['total_meals']}")
        s["current_meal_idx"] = new_idx
        # if dinner→next-lunch, advance date
        if cur_idx % 2 == 1:  # was dinner
            cur_date = dt.date.fromisoformat(s["current_date"])
            s["current_date"] = (cur_date + dt.timedelta(days=1)).isoformat()
            s["day_index"] = int(s.get("day_index", 1)) + 1
        # write
        _state_path(sid, root).write_text(...)
        return s
```

### 2. `chisha/sandbox_decision_diff.py` (新建)

```python
"""D panel 因果链派生器: eat/skip 后构造 Decision dict.

无副作用纯函数; 由 S-06c eat/skip 端点调用后落 sessions/{sid}/decisions/{idx}.json.
"""

def build_decision(
    *,
    sid: str,
    meal_idx: int,
    picked_rec: dict | None,     # None = skip
    prev_long_term_prefs: dict,  # 上一顿落盘的 long_term_prefs.json 快照
    new_long_term_prefs: dict,   # 本顿后 (eat 后已抽取) 的 long_term_prefs.json
    history: list[dict],         # 完整 history (含本顿)
    root: Path | None = None,
) -> dict:
    """返 Decision dict:
    {
      "when": "D3 午",
      "pick": "蓉香记 · 回锅肉+蒜苗炒腊肉" | "(跳过)",
      "rank": int | "—",
      "l3": int | "—",
      "diff": [
        {"kind":"add","field":"recent_dishes","value":"+ [蓉香记]"},
        {"kind":"add","field":"fatigue.蓉香记","from":"—","to":"1"},
        {"kind":"up","field":"taste.<香>","from":"0.32","to":"0.35","delta":"+0.03"},
      ],
      "implications": [...]
    }
    """
    meal_type, day = meal_idx_to_slot(meal_idx)
    when_slot = "午" if meal_type == "lunch" else "晚"
    when = f"D{day} {when_slot}"

    if picked_rec is None:
        return {
            "when": when, "pick": "(跳过)", "rank": "—", "l3": "—",
            "diff": [],
            "implications": [{"field":"—", "text":"跳过未触发学习,下一顿系统状态不变"}],
        }
    
    diff = []
    diff.append({"kind":"add", "field":"recent_dishes", "value": f"+ [{picked_rec['name']}]"})
    # fatigue: 计数 history 内同 name
    same_count = sum(1 for h in history[:-1] if h.get("dish") == picked_rec["name"])
    diff.append({"kind":"add", "field": f"fatigue.{picked_rec['name']}", 
                 "from": str(same_count) if same_count > 0 else "—",
                 "to": str(same_count + 1)})
    # taste diff: 对比 long_term_prefs.json boost 字段变化
    prev_boost = (prev_long_term_prefs or {}).get("boost", {})
    new_boost = (new_long_term_prefs or {}).get("boost", {})
    for token, v in new_boost.items():
        old_v = prev_boost.get(token, 0.0)
        if abs(v - old_v) > 1e-3:
            kind = "up" if v > old_v else "dn"
            diff.append({"kind": kind, "field": f"taste.<{token}>",
                         "from": f"{old_v:.2f}", "to": f"{v:.2f}",
                         "delta": f"{'+' if v>old_v else ''}{v-old_v:.2f}"})
    
    implications = []
    for d in diff:
        if d["kind"] in ("up", "dn") and d["field"].startswith("taste."):
            implications.append({"field": d["field"], "text": f"下顿 L2 给同类 {d['delta']} 倾向"})
        elif d["kind"] == "add" and d["field"].startswith("fatigue."):
            implications.append({"field": d["field"], "text": "下顿同菜折 0.95×"})
    
    return {
        "when": when,
        "pick": f"{picked_rec['name']} · {' + '.join(picked_rec.get('dishes',[]))}" if picked_rec.get('dishes') else picked_rec['name'],
        "rank": picked_rec.get("rank"),
        "l3": picked_rec.get("l3"),
        "diff": diff,
        "implications": implications,
    }
```
